#include "stdafx.h"
#include <iostream>
using namespace std;
int main()
{
	setlocale(LC_ALL, "");

	double x, eps;
	cout << "x = ";
	cin >> x;
	cout << "eps = ";
	cin >> eps;

	double ds = 1;
	double sum = 0;
	int i = 0;
	while (fabs(ds) > eps)
	{
		i++;
		ds *= ((4*i-3) / (4.0*i))*x;
		if (i % 2 == 1)
			sum += ds;
		else
			sum -= ds;
	}
	double f = 1 - (1.0 / (pow(1 + x, 1.0 / 4.0)));
	cout << "Найдено в ручную " << sum << endl;
	cout << "Контрольная формула " << f << endl;

	system("pause");
	return 0;
}
