void del(sportsmen result[])
{
	ifstream fin("result.txt");
	char sername[20];
	cout << "Введите фамилию для удаления записи : ";
	cin >> sername;

	char test_name[20];
	int str=0;
	while (fin >> test_name)
	{
		if (!strcmp(test_name, sername))
			break;
		fin.ignore(255, '\n');
		str++;
	}
	fin.close();
	fin.open("result.txt");
	string line;
	string text_file;
	int str_test = 0;
	while (getline(fin, line))
	{
		if (!(str == str_test))
		{
			text_file.insert(text_file.size(), line);
			text_file.insert(text_file.size(), "\r\n");
		}
		str_test++;
	}
	fin.close();
	ofstream fout;
	fout.open("result.txt", ios::trunc, ios::binary);
	fout.write(text_file.c_str(), text_file.size());
	fout.close();
}

int save(sportsmen result[], int n)
{

	for (int i = 0; i < n; i++)
	{
		fout << result[i].sername << " " ;
		fout << result[i].time_100 << " ";
		fout << result[i].time_3  << " ";
		fout << result[i].len << " ";
		fout << result[i].height << " ";
		fout << endl;
	}
	cout << "Данные сохранены!" << endl << endl;
	fout.close();
	n = 0;
	return n;
}
