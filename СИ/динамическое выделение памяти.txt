int main()
{
	setlocale(LC_ALL, "");
	int i = 0;
	char c;
	char *str;
	str = (char*)malloc(1);

	while ((c = getchar()) != '\n')
	{
		str[i] = c;
		i++;
		str = (char*)realloc(str, i + 1); // Исправил только эту строчку твоего кода
	}
	str[i] = NULL;
	_getch();
	return 0;
}
